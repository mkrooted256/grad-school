# calculate projection
def proj(point, a, b):
    l = b-a
    v = point-a
    p1 = (v @ l)/norm(l)**2 * l + a
    return p1

# calculate pseudo-coords
def calc_pc(points, M):
    #   
    sides = []
    for i in range(points.shape[0]-1):
        sides.append(points[i+1]-points[i])
    sides.append(points[0]-points[-1])
    S = np.array(sides)
    
    #        
    V = M - points
    #  ""
    T = (S*V).sum(axis=1) / norm(S, axis=1)**2
    return T
