def prepare_lagrange(xs, ys):
    n = len(xs)
    C = []
    for k in range(0, n):
        c_k = 1
        for i in range(0, n):
            if i == k:
                continue
            c_k *= (xs[k] - xs[i])
        C.append( ys[k] / c_k )
        
    return C

def evaluate_lagrange(C, xs, ys, x, eps=1e-4):
    # d = np.abs(xs - x)

    n = len(C)
    y = 0
    for k in range(0,n):
        dx = 1
        for i in range(0, n):
            if i==k:
                continue
            dx *= (x - xs[i])
        y += C[k] * dx

from numpy.polynomial import Polynomial

def T(n, a, b):
    assert a<b
    T0 = Polynomial([1])   # 1
    T1 = Polynomial([-(b+a)/(b-a),2/(b-a)]) # x = 2y/(b-a) - (b+a)/(b-a)
    if n == 0: return T0
    if n == 1: return T1
    
    return 2*T1*T(n-1, a, b) - T(n-2, a, b)
